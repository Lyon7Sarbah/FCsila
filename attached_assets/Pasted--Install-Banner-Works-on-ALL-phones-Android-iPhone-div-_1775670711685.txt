<script>
    // Show banner after 2 seconds
    setTimeout(() => {
        document.getElementById('installBanner').style.transform = 'translateY(0)';
    }, 2000);
    
    // Close banner
    document.getElementById('closeBanner').onclick = () => {
        document.getElementById('installBanner').style.transform = 'translateY(150%)';
    };
    
    // Show iPhone instructions modal
    document.getElementById('showIphoneInstructions').onclick = () => {
        document.getElementById('iphoneModal').style.display = 'flex';
    };
    
    // Close modal
    document.getElementById('closeModal').onclick = () => {
        document.getElementById('iphoneModal').style.display = 'none';
    };
    
    // Close modal when clicking outside
    document.getElementById('iphoneModal').onclick = (e) => {
        if (e.target === document.getElementById('iphoneModal')) {
            document.getElementById('iphoneModal').style.display = 'none';
        }
    };
    
    // Android automatic install prompt
    let deferredPrompt;
    window.addEventListener('beforeinstallprompt', (e) => {
        e.preventDefault();
        deferredPrompt = e;
        document.getElementById('installAndroidBtn').style.display = 'block';
        
        document.getElementById('installAndroidBtn').onclick = () => {
            deferredPrompt.prompt();
            deferredPrompt.userChoice.then((result) => {
                if (result.outcome === 'accepted') {
                    console.log('User installed app');
                }
                deferredPrompt = null;
                document.getElementById('installAndroidBtn').style.display = 'none';
                document.getElementById('installBanner').style.transform = 'translateY(150%)';
            });
        };
    });
    
    // Detect if user is on iPhone and show different message
    const isIPhone = /iPhone/i.test(navigator.userAgent);
    const isAndroid = /Android/i.test(navigator.userAgent);
    
    if (isIPhone) {
        document.getElementById('installMessage').innerHTML = '📱 Add FC SILA to your iPhone home screen!';
    } else if (isAndroid) {
        document.getElementById('installMessage').innerHTML = '📱 Install FC SILA as an app on your Android!';
    }
</script>
